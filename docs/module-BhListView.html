BhListView

使用vue组件来自定义item的listview
Source:
  • bh-listview/bhListview.vue, line 2
Examples

html

<list-view :page-path='pagePath'
     :params='options.params'
     :item-comp='compItem'>
     <div class='empty-page'>
         列表为空的时候展示这句话
     </div>
 </list-view>

javascript

 // index.js
 import TaskItem from './taskItem.vue';
 export default {
     props: {
         compItem: TaskItem
     }
 }
 // taskItem.vue
 export default {
     props: {
         item: { // 此属性为每行数据的对象，可在template中随意渲染
             type: Object,
             default: () => ({})
         }
     }
 }
